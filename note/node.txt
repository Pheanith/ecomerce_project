// register route
app.get("/",(req,res)=>{
    res.send("You have request my root server")
})
// res.query, res.param
app.get("/api/user",(req,res)=>{
    var obj = {
        list1 : [10,20],
        list2 : ["Sok", "Som"] 
    }
    res.json(obj)
})
// req.query
app.get("/api/product",(req,res)=>{
    res.json({
        a:10,
        b:20,
        getQuery:req.query
    })
})
// req.params
app.get("/api/product/:id",(req,res)=>{
    var id = req.params.id
    var arrProduct = [
        {
            name: "coca",
            id: 101,
            price: 1
        },
        {
            name: "lala",
            id: 102,
            price: 2
        },
        {
            name: "Dara",
            id: 103,
            price: 3
        },
    ]

    var productTmp = []
    for (var i = 0; i < arrProduct.length ; i++){
        if(arrProduct[i].id == id){
            productTmp.push(arrProduct[i])
            break;
        }
    }
    res.json({
        List: productTmp
    })
})
// req.body
app.post("/api/product",(req,res)=>{
    res.json({
        message: req.body
    })
})
